class Event {
   constructor(name, time, description) {
     this.name = name;
     this.time = time;
     this.description = description;
} }
// Function to notify users of an upcoming event
function notifyEvent(event) {
   console.log(`Event: ${event.name}\nTime: ${event.time}\nDescription: ${event.description}\n`);
}
// Function to create a timer for an event
function createEventTimer(event) {
   const eventTime = new Date(event.time);
   const currentTime = new Date();
   const timeDifference = eventTime.getTime() - currentTime.getTime();
   setTimeout(() => {
     notifyEvent(event);
}, timeDifference);
}
